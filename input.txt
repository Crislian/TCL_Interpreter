proc fib { {n} } {
	
	#puts $n;
  if { $n == 1 || $n == 0 } then {
    return 1;
  }
  set a [fib { expr { $n - 1 } } ];
  
	puts "a = $a";
  set b [fib { expr { $n - 2 } } ];
	
	puts "b = $b";

  return [ expr { $a + $b} ];
}

for {set i 0}{ $i < 10 }{ incr i} {
    puts "fib $i";
    puts [fib {$i} ];
}
